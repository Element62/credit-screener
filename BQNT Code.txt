    "    Returns clean DataFrame [ID, name], one row per ID.\n",
    "    Handles multi-column responses by stripping all known noise columns.\n",
    "    Returns empty DataFrame on any failure.\n",

    "    try:\n",

    "        ({'NAME'           : bq.data.name()},                     'NAME'),\n",
    "    ]\n",
    "\n",
    "    for field_dict, name in fields:\n",
    "        df_field = _pull_field(universe, field_dict, name)\n",
    "        if not df_field.empty:\n",
    "            df = df.merge(df_field, on='ID', how='left')\n",
    "        else:\n",
    "            df[name] = pd.NA\n",
    "\n",
    "    return df\n",
    "\n",
    "\n",

    "def build_issuer_table(df_issuer_analytics, df_instrument_analytics, df_names, df_sectors):\n",
    "    df = df_issuer_analytics.copy()\n",
    "\n",
    "    # tranche count\n",
    "    tranche_count = (\n",
    "        df_instrument_analytics\n",
    "        .groupby('PARENT_TICKER')['ID']\n",
    "        .count()\n",
    "        .rename('# Tranches')\n",
    "        .reset_index()\n",
    "    )\n",
    "\n",
    "    # nearest upcoming maturity\n",
    "    today = pd.Timestamp.today()\n",
    "    nearest_mat = (\n",
    "        df_instrument_analytics\n",
    "        .assign(MAT_DT=pd.to_datetime(df_instrument_analytics['MATURITY'], errors='coerce'))\n",
    "        .query('MAT_DT >= @today')\n",
    "        .groupby('PARENT_TICKER')['MAT_DT']\n",
    "        .min()\n",
    "        .dt.strftime('%Y-%m-%d')\n",
    "        .rename('Nearest Maturity')\n",
    "        .reset_index()\n",
    "    )\n",
    "\n",
    "    df = (\n",
    "        df\n",
    "        .merge(df_names,       on='PARENT_TICKER', how='left')\n",
    "        .merge(df_sectors,     on='PARENT_TICKER', how='left')\n",
    "        .merge(tranche_count,  on='PARENT_TICKER', how='left')\n",
    "        .merge(nearest_mat,    on='PARENT_TICKER', how='left')\n",
    "    )\n",
    "\n",
    "    # maturity wall labels\n",
    "    def _wall_label(mm, pct):\n",
    "        if pd.isna(mm) or mm == 0:\n",
    "            return '-'\n",
    "        return f\"${mm/1000:.2f}B | {pct:.0f}%\"\n",
    "\n",
    "    df['<1Y']  = df.apply(lambda r: _wall_label(r.get('WALL_LT1Y_MM', 0),  r.get('WALL_LT1Y_PCT', 0)),  axis=1)\n",
    "    df['1-3Y'] = df.apply(lambda r: _wall_label(r.get('WALL_1_3Y_MM', 0),  r.get('WALL_1_3Y_PCT', 0)),  axis=1)\n",
    "    df['3-5Y'] = df.apply(lambda r: _wall_label(r.get('WALL_3_5Y_MM', 0),  r.get('WALL_3_5Y_PCT', 0)),  axis=1)\n",
    "\n",
    "    # rename to display names\n",
    "    df = df.rename(columns={\n",
    "        'PARENT_TICKER'                  : 'Issuer',\n",
    "        'ISSUER_FACE_MM'                 : 'Face ($MM)',\n",
    "        'ISSUER_WTAVG_PX'                : 'WA Price',\n",
    "        'ISSUER_DIST_TO_PAR'             : 'Dist to Par (pts)',\n",
    "        'ISSUER_DISLOCATION_MM'          : 'Par Upside ($MM)',\n",
    "        'ISSUER_DISLOCATION_52W_MM'      : '52W Peak Upside ($MM)',\n",
    "        'ISSUER_DISLOCATION_52W_SEC_MM'  : '52W Secured Upside ($MM)',\n",
    "        'ISSUER_DISLOCATION_52W_UNSEC_MM': '52W Unsecured Upside ($MM)',\n",
    "        'SECURED_PCT'                    : 'Secured (%)',\n",
    "        'SENIORITY_GAP'                  : 'Seniority Basis (pts)',\n",
    "        'GAP_SIGNAL'                     : 'Gap Signal',\n",
    "        'ISSUER_OAS_DELTA'               : 'OAS Delta (bps)',\n",
    "        'SR_UNSEC_EXCESS_OAS_DELTA'      : 'Adj Unsecured Sprd Movement (bps)',\n",
    "        'SR_UNSEC_MOM_LABEL'             : 'Sprd Movement Label',\n",
    "    })\n",
    "\n",
    "    df['Face ($BN)'] = df['Face ($MM)'] / 1000.0\n",
    "\n",
    "    col_order = [\n",
    "        'Issuer', 'Issuer Name', 'Sector',\n",
    "        'Face ($BN)',\n",
    "        'WA Price', 'Dist to Par (pts)',\n",
    "        'Par Upside ($MM)',\n",
    "        '52W Peak Upside ($MM)',\n",
    "        '52W Secured Upside ($MM)',\n",
    "        '52W Unsecured Upside ($MM)',\n",
    "        'Secured (%)',\n",
    "        'Seniority Basis (pts)', 'Gap Signal',\n",
    "        '<1Y', '1-3Y', '3-5Y', 'Nearest Maturity',\n",
    "        'OAS Delta (bps)',\n",
    "        'Adj Unsecured Sprd Movement (bps)', 'Sprd Movement Label',\n",
    "        '# Tranches',\n",
    "        'Face ($MM)',\n",
    "    ]\n",
    "    df = df[[c for c in col_order if c in df.columns]]\n",
    "\n",
    "    print(f\"build_issuer_table: {len(df)} issuers ready.\")\n",
    "    return df\n",

    "def build_tranche_sections(issuer_ticker):\n",
    "    df_instr = df_instrument_analytics[\n",
    "        df_instrument_analytics['PARENT_TICKER'] == issuer_ticker\n",
    "    ].copy()\n",
    "\n",
    "    if df_instr.empty:\n",
    "        return widgets.HTML(\"<p style='color:#aaa'>No instrument data.</p>\")\n",
    "\n",
    "    df_instr['RANK_LABEL'] = df_instr['PAYMENT_RANK'].fillna('Unranked').astype(str)\n",
    "    df_instr['RANK_ORDER'] = df_instr['RANK_LABEL'].map(SENIORITY_ORDER).fillna(MAX_KNOWN_RANK + 1)\n",
    "\n",
    "    total_face = pd.to_numeric(df_instr['AMT_OUTSTANDING'], errors='coerce').sum() / 1e6\n",
    "\n",
    "    # format rows\n",
    "    rows = df_instr.copy()\n",
    "    rows['AMT Outstanding ($MM)'] = pd.to_numeric(rows['AMT_OUTSTANDING'], errors='coerce') / 1e6\n",
    "    rows['Weight (%)']            = (rows['AMT Outstanding ($MM)'] / total_face * 100).round(1)\n",
    "    rows['PX Mid']                = pd.to_numeric(rows['PX_MID'],            errors='coerce').round(2)\n",
    "    rows['52W High']              = pd.to_numeric(rows['PX_HIGH_52W'],       errors='coerce').round(2)\n",
    "    rows['vs 52W High (pts)']     = pd.to_numeric(rows['DIST_TO_52W_HIGH'],  errors='coerce').round(2)\n",
    "    rows['Dist to Par (pts)']     = pd.to_numeric(rows['DIST_TO_PAR'],       errors='coerce').round(2)\n",
    "    rows['Par Upside ($MM)']      = pd.to_numeric(rows['DISLOCATION_MM'],    errors='coerce').round(2)\n",
    "    rows['52W Peak Upside ($MM)'] = pd.to_numeric(rows['DISLOCATION_52W_MM'],errors='coerce').round(2)\n",
    "    rows['Distress Tier']         = rows['DISTRESS_TIER'].fillna('')\n",
    "    rows['Payment Rank']          = rows['RANK_LABEL']\n",
    "    rows['OAS']                   = pd.to_numeric(rows['OAS'],               errors='coerce').round(2)\n",
    "    rows['OAS Delta (bps)']       = pd.to_numeric(rows['OAS_DELTA'],         errors='coerce').round(2)\n",
    "    rows['Maturity']              = pd.to_datetime(rows['MATURITY'], errors='coerce').dt.strftime('%Y-%m-%d')\n",
    "    rows['Adj Sprd Movement (bps)'] = pd.to_numeric(\n",
    "        rows.get('SR_UNSEC_EXCESS_OAS_DELTA', pd.NA), errors='coerce').round(2)\n",
    "    rows['Sprd Movement Label']   = rows.get('SR_UNSEC_MOM_LABEL', '').fillna('')\n",
    "\n",
    "    sections    = []\n",
    "    rank_groups = sorted(rows['RANK_ORDER'].dropna().unique().tolist())\n",
    "\n",
    "    def _render_group(label, group_df):\n",
    "        t_face = group_df['AMT Outstanding ($MM)'].sum()\n",
    "        t_par  = group_df['Par Upside ($MM)'].sum()\n",
    "        t_52w  = group_df['52W Peak Upside ($MM)'].sum()\n",
    "        pct    = t_face / total_face * 100 if total_face > 0 else 0\n",
    "        color  = SENIORITY_COLORS.get(label, '#7f8c8d')\n",
    "\n",
    "        header_html = widgets.HTML(\n",
    "            f\"<div style='background:rgb(70,50,20); padding:6px 12px;\"\n",
    "            f\"border-left:4px solid {color}; margin-top:10px; border-radius:2px;'>\"\n",
    "            f\"<span style='color:{color}; font-weight:bold; font-size:13px'>▶  {label}</span>\"\n",
    "            f\"<span style='color:#aaaaaa; font-size:12px; margin-left:20px'>${t_face:,.1f}MM face</span>\"\n",
    "            f\"<span style='color:#aaaaaa; font-size:12px; margin-left:12px'>{pct:.1f}% of stack</span>\"\n",
    "            f\"<span style='color:#e15759; font-size:12px; margin-left:12px'>Par Upside: ${t_par:,.1f}MM</span>\"\n",
    "            f\"<span style='color:#f0ad4e; font-size:12px; margin-left:12px'>52W Upside: ${t_52w:,.1f}MM</span>\"\n",
    "            f\"</div>\"\n",
    "        )\n",
    "        sorted_df = group_df.sort_values('Par Upside ($MM)', ascending=False).reset_index(drop=True)\n",
    "        sections.append(widgets.VBox([header_html, build_tranche_grid(sorted_df)]))\n",
    "\n",
    "    for rank_order in rank_groups:\n",
    "        group = rows[rows['RANK_ORDER'] == rank_order].copy()\n",
    "        _render_group(group['RANK_LABEL'].iloc[0], group)\n",
    "\n",
    "    if rows['RANK_ORDER'].isna().any():\n",
    "        _render_group('Unranked', rows[rows['RANK_ORDER'].isna()].copy())\n",
    "\n",
    "    return widgets.VBox(sections)\n",
